/**
 * 两个数的加法
 * 同样忽略了小数点
 */
string add(string a, string b) {
  string v;
  int borrow = 0;
  for(int i = a.size() - 1, j = b.size() - 1; i >= 0 || j >= 0; --i, --j) {
    int cur = (i >= 0 ? a[i] - '0' : 0) + (j >= 0 ? b[j] - '0' : 0) + borrow;
    v += '0' + cur % 10;
    borrow = cur / 10;
  }
  if(borrow) v += '0' + borrow;
  reverse(v.begin(), v.end());
  return v;
}
